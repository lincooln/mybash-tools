# Справка по командам
ls ls -a # показать все файлы (включая скрытые)
ls ls -lh # подробный список с размерами
grep grep -i # игнорировать регистр
grep grep -r # рекурсивный поиск в папке
grep grep -n # показать номера строк
tar tar -czf архив.tar.gz папка/ # создать сжатый архив
tar tar -xzf архив.tar.gz # распаковать сжатый архив
systemctl systemctl start имя # запустить службу
systemctl systemctl status имя # проверить статус службы

# Справка по vi/vim
vi start # Чтобы начать: нажми i → печатай → ESC → :wq (сохранить) или :q! (выйти без сохранения)
vi panic # Застрял? Нажми ESC несколько раз → введи :q! → Enter
vi :q # Выйти (если ничего не меняли)
vi :q! # Выйти без сохранения
vi :wq # Сохранить и выйти
vi i # Перейти в режим вставки (можно печатать)
vi ESC # Выйти из режима вставки (вернуться в командный режим)
vi x # Удалить символ под курсором
vi dd # Удалить строку
vi /слово # Найти "слово" в тексте
vi :set nu # Показать номера строк

# Клавиатурные сочетания
key Ctrl+A # Перейти в начало строки
key Ctrl+E # Перейти в конец строки
key Ctrl+U # Удалить всё до курсора
key Ctrl+K # Удалить всё после курсора
key Ctrl+W # Удалить слово перед курсором
key Ctrl+R # Поиск в истории команд
key Alt+. # Вставить последний аргумент предыдущей команды
key Tab # Автодополнение команды или файла
